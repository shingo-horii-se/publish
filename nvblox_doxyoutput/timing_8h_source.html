58template <typename T, typename Total, int N>
59class Accumulator {
60 public:
61 Accumulator()
62 : window_samples_(0),
63 totalsamples_(0),
64 window_sum_(0),
65 sum_(0),
66 min_(std::numeric_limits<T>::max()),
67 max_(std::numeric_limits<T>::min()) {}
68
69 void Add(T sample) {
70 if (window_samples_ < N) {
71 samples_[window_samples_++] = sample;
72 window_sum_ += sample;
73 } else {
74 T& oldest = samples_[window_samples_++ % N];
75 window_sum_ += sample - oldest;
76 oldest = sample;
77 }
78 sum_ += sample;
79 ++totalsamples_;
80 if (sample > max_) {
81 max_ = sample;
82 }
83 if (sample < min_) {
84 min_ = sample;
85 }
86 }
87
88 int TotalSamples() const { return totalsamples_; }
89
90 double Sum() const { return sum_; }
91
92 double Mean() const { return sum_ / totalsamples_; }
93
94 double RollingMean() const {
95 return window_sum_ / std::min(window_samples_, N);
96 }
97
98 double Max() const { return max_; }
99
100 double Min() const { return min_; }
101
102 double LazyVariance() const {
103 if (window_samples_ == 0) {
104 return 0.0;
105 }
106 double var = 0;
107 double mean = RollingMean();
108 for (int i = 0; i < std::min(window_samples_, N); ++i) {
109 var += (samples_[i] - mean) * (samples_[i] - mean);
110 }
111 var /= std::min(window_samples_, N);
112 return var;
113 }
114
115 private:
116 int window_samples_;
117 int totalsamples_;
118 Total window_sum_;
119 Total sum_;
120 T min_;
121 T max_;
122 T samples_[N];
123};
