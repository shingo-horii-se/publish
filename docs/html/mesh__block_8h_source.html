97template <>
98inline BlockLayer<MeshBlock>::BlockLayer(const BlockLayer& other,
99 MemoryType memory_type)
100 : BlockLayer(other.block_size_, memory_type) {
101 LOG(INFO) << "Deep copy of Mesh BlockLayer containing "
102 << other.numAllocatedBlocks() << " blocks.";
103 // Re-create all the blocks.
104 std::vector<Index3D> all_block_indices = other.getAllBlockIndices();
105
106 // Iterate over all blocks, clonin'.
107 for (const Index3D& block_index : all_block_indices) {
108 typename MeshBlock::ConstPtr block = other.getBlockAtIndex(block_index);
109 if (block == nullptr) {
110 continue;
111 }
112 blocks_.emplace(block_index,
113 std::make_shared<MeshBlock>(*block, memory_type));
114 }
115}
