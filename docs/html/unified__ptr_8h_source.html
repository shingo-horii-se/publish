154// These are used to specialize make_unique on the return type, which is need to
155// generate class templates for non-array and array types. See gcc here:
156// https://github.com/gcc-mirror/gcc/blob/master/libstdc%2B%2B-v3/include/bits/unique_ptr.h#L943
157template <class T>
158struct _Unified_if {
159 typedef unified_ptr<T> _Single_object;
160};
161
162template <class T>
163struct _Unified_if<T[]> {
164 typedef unified_ptr<T[]> _Unknown_bound;
165};
166
167template <class T, size_t N>
168struct _Unified_if<T[N]> {
169 typedef void _Known_bound;
170};

172// Single object, default storage location
